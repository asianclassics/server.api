# REACT BUILD, with EXPRESS backend, proxied by NGINX

server {
    
    listen      80;
    listen [::]:80;

    server_name catalogs.asianclassics.org;

    # root directory for index.html, starting point for UI
    location / {
        root /var/www/html;
        index  index.html index.htm;
    }

    location /api {
        proxy_redirect off;
        proxy_set_header   X-Real-IP            $remote_addr;
        proxy_set_header   X-Forwarded-For      $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto    $scheme;
        proxy_set_header   Host                 $http_host;
        proxy_set_header   X-NginX-Proxy        true;
        proxy_set_header   Connection           "";
        proxy_http_version                      1.1;
        proxy_cache                             one;
        proxy_cache_key                         sfs$request_uri$scheme;
        rewrite ^/api/?(.*) /$1 break;
        proxy_pass                              http://express_upstream;
    }

    # serving static files through NGINX
    location ~ ^/(manifest.json|images/|img/|javascript/|js/|css/|stylesheets/|media/|static/|robots.txt|humans.txt|favicon.ico) {
        root /var/www/html;
        access_log off;
        expires max;
    }

}
